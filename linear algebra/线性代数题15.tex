\begin{exercise}\footnote{来自 http://www.math.ucla.edu/~tao/resource/general/115a.3.02f/final.pdf}
设 $V$ 是一个实向量空间,且 $T:V\to V$ 是一个线性变换,满足 $T^2=T$.
\begin{itemize}
\item 证明 $R(T)\cap N(T)=\{0\}$.
\item 
\end{itemize}

\end{exercise}
\begin{proof}[\textbf{证明}]
  \begin{itemize}
  \item 假若存在 $v\neq \mathbf{0}$,$v\in R(T)\cap N(T)$,则必存在
    $w\in V$,使得$T(w)=v$.易得 $w\neq \mathbf{0}$(否则
    $T(w)=\mathbf{0}\neq v$).且 $T(v)=T^2(w)=T(w)=0$.矛盾.
  \end{itemize}
\end{proof}
